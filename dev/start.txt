1. Run docker
2. RUN DOCKER CONTAINER (--rm to remove when stopped):
docker run --name bim-dev -e POSTGRES_PASSWORD=admin -p 5432:5432 -d --rm postgres
3. PGadmin for DB UI
Create db 'bim-dev'



CREATE DATABASE:
'bim-dev'

POSTGRES DB OPTIONS ( 'server/config/index.ts': POSTGRES_OPTIONS ):
host: 'localhost',
port: 5432,
database: 'bim-dev',
dialect: 'postgres',
username: 'postgres',
password: 'admin',

BE:
npm run start
